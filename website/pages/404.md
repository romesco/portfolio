    // > not-found). The tracker also logs a normal pageview for the bad URL;
    // this makes every 404 findable in one place regardless of path.
    (function () {
      function fire() { try { umami.track('not-found', { path: location.pathname + location.search }); } catch (e) {} }
      if (window.umami) fire();
      else { var n = 0, t = setInterval(function () { if (window.umami) { clearInterval(t); fire(); } else if (++n > 20) { clearInterval(t); } }, 400); }
    })();
  </script>
